# Quick start

Get your synchronized radio streaming in under 5 minutes. This guide will help you set up the essential files and start broadcasting.

## Prerequisites

Before you begin, make sure you have:

- A web server (Apache, Nginx, or local development server)
- Audio files in MP3 format
- Basic knowledge of HTML and JavaScript

<Note>
You can use Python's built-in server for local testing: `python3 -m http.server 8000`
</Note>

  <Step title="Set up the playlist">
    Create a `playlist.json` file with your audio tracks:

    ```json
    { "tracks": [
      "music/track1.mp3",
      "music/track2.mp3",
      "music/track3.mp3"
    ] }
    ```

    <Note>
    Make sure your audio files are accessible relative to the HTML file. Create a `music/` directory and add your MP3 files there.
    </Note>
  </Step>

  <Step title="Add the radio script">
    Create `radio-zara.js` with the core synchronization logic:

    ```javascript
    document.addEventListener('DOMContentLoaded', function() {
        const playButton = document.getElementById('radioPlayButton');
        let audioPlayer = document.getElementById('radioPlayer');
        const playPath = document.getElementById('playPath');
        const pausePath1 = document.getElementById('pausePath1');
        const pausePath2 = document.getElementById('pausePath2');
        
        let isPlaying = false;
        let currentPlaylist = [];
        let currentTrackIndex = 0;
        let playlistLoaded = false;

        // Load playlist from JSON
        async function loadPlaylist() {
            if (playlistLoaded) return;
            
            try {
                console.log('📻 Loading playlist...');
                const response = await fetch('playlist.json');
                const data = await response.json();
                
                currentPlaylist = data.tracks.map(track => ({
                    path: track,
                    file: track.split('/').pop()
                }));
                
                playlistLoaded = true;
                console.log(`📻 Playlist loaded: ${currentPlaylist.length} tracks`);
                
            } catch (error) {
                console.error('Error:', error);
                currentPlaylist = [];
                currentTrackIndex = 0;
            }
        }

        // Calculate exact synchronized position
        function calcularPosicionExacta() {
            const inicioTransmision = new Date('2025-01-01T03:00:00Z');
            const ahora = new Date();
            
            const segundosTranscurridos = Math.floor((ahora - inicioTransmision) / 1000);
            const segundosPorCancion = 180; // 3 minutes per track
            const segundosTotalPlaylist = currentPlaylist.length * segundosPorCancion;
            const posicionEnPlaylist = segundosTranscurridos % segundosTotalPlaylist;
            
            currentTrackIndex = Math.floor(posicionEnPlaylist / segundosPorCancion) % currentPlaylist.length;
            const segundoEnCancion = posicionEnPlaylist % segundosPorCancion;
            
            return {
                trackIndex: currentTrackIndex,
                segundoEnCancion: segundoEnCancion,
                track: currentPlaylist[currentTrackIndex]
            };
        }

        // Play synchronized transmission
        function playTransmisionExacta() {
            if (currentPlaylist.length === 0) return;
            
            const posicion = calcularPosicionExacta();
            const track = posicion.track;
            
            console.log(`🎵 Playing: "${track.file}"`);
            console.log(`🎯 Starting at second: ${posicion.segundoEnCancion}`);
            
            audioPlayer.src = track.path;
            audioPlayer.currentTime = Math.min(posicion.segundoEnCancion, 3600);
            
            audioPlayer.play().catch(e => {
                console.error('Playback error:', e);
            });
            
            audioPlayer.onended = function() {
                siguienteCancion();
            };
        }

        // Next track
        function siguienteCancion() {
            if (currentPlaylist.length === 0) return;
            
            currentTrackIndex = (currentTrackIndex + 1) % currentPlaylist.length;
            const track = currentPlaylist[currentTrackIndex];
            
            audioPlayer.src = track.path;
            audioPlayer.currentTime = 0;
            
            if (isPlaying) {
                audioPlayer.play();
            }
            
            audioPlayer.onended = function() {
                siguienteCancion();
            };
        }

        // Update play button state
        function updatePlayButton() {
            playPath.setAttribute('opacity', isPlaying ? '0' : '1');
            pausePath1.setAttribute('opacity', isPlaying ? '1' : '0');
            pausePath2.setAttribute('opacity', isPlaying ? '1' : '0');
        }

        // Play/Pause event
        playButton.addEventListener('click', async function() {
            if (isPlaying) {
                audioPlayer.pause();
                isPlaying = false;
                console.log('⏸️ Paused');
            } else {
                if (!playlistLoaded) {
                    await loadPlaylist();
                }
                isPlaying = true;
                console.log('▶️ Starting synchronized transmission...');
                
                playTransmisionExacta();
            }
            updatePlayButton();
        });

        // Initialize
        loadPlaylist();
        console.log('✅ Radio ready');
    });
    ```

## Verify synchronization

To test that synchronization is working:

1. Open your radio in two different browser tabs
2. Click play in both tabs at slightly different times
3. Both players should jump to the same position in the same track
4. All listeners hear exactly the same content at the same time

<CodeGroup>
```bash Terminal 1
# Start first browser instance
open http://localhost:8000
```

```bash Terminal 2

open http://localhost:8000
```
</CodeGroup>
